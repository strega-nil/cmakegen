cmake_minimum_required(VERSION 3.6)

project($projname VERSION 0.1 LANGUAGES CXX)

add_library($projname INTERFACE)

target_include_directories($projname
  INTERFACE
    $$<BUILD_INTERFACE:$${CMAKE_CURRENT_SOURCE_DIR}/include>
    $$<INSTALL_INTERFACE:include>)

target_compile_features($projname INTERFACE cxx_std_$cxx_version)

if(MSVC)
  target_compile_options($projname
    INTERFACE
      /D_SCL_SECURE_NO_WARNINGS
      /permissive-)
endif()
